  def arr_dist(collapse = true, res=10)
    dist = Hash.new
    @proteins.values.each do |p|
      p = p.collapse if (collapse)
      next unless p.has_domains?
      d_no = p.domains.size
      if (d_no <= res)
        dist[d_no] = (dist.has_key?(d_no)) ? dist[d_no].succ : 1
      else
        k = ">#{res}"
        dist[k] = (dist.has_key?(k)) ? dist[k].succ : 1
      end
    end
    return dist
  end
