  def find_uniq_arr_by_dom(did, collapse=true)
    uniq = Hash.new
    prots = self.find_arr_by_dom(did)
    return nil if prots.nil?
    prots.each do |p|
      p = p.collapse if collapse
      next if uniq.has_key?(p.arr_str)
      uniq[p.arr_str] = p
    end
    return uniq.values
  end
